=> Helper functions {{{

function! CmdLine(str)
    exe "menu Foo.Bar :. a:str
    emenu Foo.Bar
    unmenu Foo
endfunction 

function! VisualSelection(direction, extra_filter) range
    let l:saved_reg = @"
    execute "normal! vgvy"

    let l:pattern = escape(@", '\\/.*$^~[]')
    let l:pattern = substitute(l:pattern, "\n$", , )

    if a:direction == 'b'
        execute "normal ?. l:pattern . "^M"
    elseif a:direction == 'gv'
        call CmdLine("Ack \. l:pattern . "\)
    elseif a:direction == 'replace'
        call CmdLine("%s. '/'. l:pattern . '/')
    elseif a:direction == 'f'
        execute "normal /. l:pattern . "^M"
    endif

    let @/ = l:pattern
    let @= l:saved_reg
endfunction

Returns true if paste mode is enabled function! HasPaste() if &paste return 'PASTE MODE ' en return '' endfunction

Don't close window, when deleting a buffer command! Bclose call BufcloseCloseIt() function! BufcloseCloseIt() let l:currentBufNum = bufnr("%") let l:alternateBufNum = bufnr("#")

   if buflisted(l:alternateBufNum)
     buffer #
   else
     bnext
   endif

   if bufnr("%") == l:currentBufNum
     new
   endif

   if buflisted(l:currentBufNum)
     execute("bdelete! ".l:currentBufNum)
   endif
endfunction
